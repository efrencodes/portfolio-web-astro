```git
[user]
	name = name
	email = mail
[alias]
	br = branch -a
	a = add .
	i = init
	s = status
	sh = stash
	shs = stash save
	shl = stash list
	shp = stash pop
	shd = stash drop
	ps = push origin main
	pl = pull origin main
	psd = push origin development
	pld = pull origin development
	l = "log --graph --abbrev-commit --date=relative --all --format=format:'%C(bold blue)%h%C(reset) %C(bold green)(%>(18,trunc)%aD)%C(reset) %C(bold blue)[%<(10,trunc)%aN]%Creset - %s%C(reset) %C(bold yellow)%d%C(reset)'"
[filter "lfs"]
	process = git-lfs filter-process
	required = true
	clean = git-lfs clean -- %f
	smudge = git-lfs smudge -- %f
[color]
	ui = auto
[init]
	defaultBranch = main
```
